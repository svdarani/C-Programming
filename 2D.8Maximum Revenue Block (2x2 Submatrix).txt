#include <stdio.h>

int main() {
    int R, C;
    scanf("%d %d", &R, &C);
    
    int mat[R][C];
    for (int i = 0; i < R; i++)
        for (int j = 0; j < C; j++)
            scanf("%d", &mat[i][j]);
    
    int maxSum = 0;
    int firstSum = mat[0][0] + mat[0][1] + mat[1][0] + mat[1][1];
    maxSum = firstSum;
    
    for (int i = 0; i <= R - 2; i++) {
        for (int j = 0; j <= C - 2; j++) {
            int sum = mat[i][j] + mat[i][j+1] + mat[i+1][j] + mat[i+1][j+1];
            if (sum > maxSum) maxSum = sum;
        }
    }
    
    printf("%d\n", maxSum);
    return 0;
}
