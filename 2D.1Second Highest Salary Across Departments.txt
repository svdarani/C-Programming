#include <stdio.h>

int main() {
    int R, C;
    scanf("%d %d", &R, &C);
    
    int salary[R][C];
    for (int i = 0; i < R; i++)
        for (int j = 0; j < C; j++)
            scanf("%d", &salary[i][j]);
    
    int highest = -1, second = -1;
    
    for (int i = 0; i < R; i++) {
        for (int j = 0; j < C; j++) {
            if (salary[i][j] > highest) {
                second = highest;
                highest = salary[i][j];
            } else if (salary[i][j] > second && salary[i][j] != highest) {
                second = salary[i][j];
            }
        }
    }
    
    printf("%d\n", second);
    return 0;
}
