#include <stdio.h>

int main() {
    int R, C;
    scanf("%d %d", &R, &C);
    
    int mat[R][C];
    for (int i = 0; i < R; i++)
        for (int j = 0; j < C; j++)
            scanf("%d", &mat[i][j]);
    
    int count = 0;
    
    for (int i = 0; i < R; i++) {
        int evenCount = 0, oddCount = 0;
        for (int j = 0; j < C; j++) {
            if (mat[i][j] % 2 == 0)
                evenCount++;
            else
                oddCount++;
        }
        if (evenCount == oddCount) count++;
    }
    
    printf("%d\n", count);
    return 0;
}
